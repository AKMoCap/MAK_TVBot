Setup Steps:
  1. Create your indicator or strategy in TradingView Pine Script
  2. Set up an alert on the chart with your indicator
  3. In the alert settings, select "Webhook URL" and paste the URL above
  4. In the "Message" field, paste the JSON template
  5. Replace values as needed:
    • YOUR_WEBHOOK_SECRET - Your secret key from Settings
    • leverage - Desired leverage (1-20)
    • collateral_usd - Amount to risk per trade
    • indicator - The webhook key for this indicator
  6. Use Pine Script placeholders:
    • {% raw %}{{strategy.order.action}}{% endraw %} - "buy" or "sell"
    • {% raw %}{{ticker}}{% endraw %} - The coin symbol
    • {% raw %}{{close}}{% endraw %} - Current price
Pro Tip: Start with TESTNET enabled to verify your setup works correctly before trading with real funds.
Sample Pine Scripts
//@version=5
strategy("RSI Strategy for MAK Bot", overlay=true)

// Parameters
rsiLength = input.int(14, "RSI Length")
oversold = input.int(30, "Oversold Level")
overbought = input.int(70, "Overbought Level")

// Calculate RSI
rsiValue = ta.rsi(close, rsiLength)

// Generate signals
longCondition = ta.crossover(rsiValue, oversold)
shortCondition = ta.crossunder(rsiValue, overbought)

if (longCondition)
    strategy.entry("Long", strategy.long)
    alert('{"secret":"YOUR_SECRET","action":"buy","coin":"' + syminfo.ticker + '","leverage":10,"collateral_usd":100,"indicator":"rsi-oversold"}', alert.freq_once_per_bar)

if (shortCondition)
    strategy.entry("Short", strategy.short)
    alert('{"secret":"YOUR_SECRET","action":"sell","coin":"' + syminfo.ticker + '","leverage":10,"collateral_usd":100,"indicator":"rsi-overbought"}', alert.freq_once_per_bar)
//@version=5
strategy("EMA Crossover for MAK Bot", overlay=true)

// Parameters
fastLength = input.int(9, "Fast EMA")
slowLength = input.int(21, "Slow EMA")

// Calculate EMAs
fastEMA = ta.ema(close, fastLength)
slowEMA = ta.ema(close, slowLength)

// Plot EMAs
plot(fastEMA, color=color.blue, title="Fast EMA")
plot(slowEMA, color=color.red, title="Slow EMA")

// Generate signals
longCondition = ta.crossover(fastEMA, slowEMA)
shortCondition = ta.crossunder(fastEMA, slowEMA)

if (longCondition)
    strategy.entry("Long", strategy.long)
    alert('{"secret":"YOUR_SECRET","action":"buy","coin":"' + syminfo.ticker + '","leverage":10,"collateral_usd":100,"indicator":"ema-cross"}', alert.freq_once_per_bar)

if (shortCondition)
    strategy.entry("Short", strategy.short)
    alert('{"secret":"YOUR_SECRET","action":"sell","coin":"' + syminfo.ticker + '","leverage":10,"collateral_usd":100,"indicator":"ema-cross"}', alert.freq_once_per_bar)
//@version=5
strategy("MACD Strategy for MAK Bot", overlay=false)

// Parameters
fastLength = input.int(12, "Fast Length")
slowLength = input.int(26, "Slow Length")
signalLength = input.int(9, "Signal Length")

// Calculate MACD
[macdLine, signalLine, histLine] = ta.macd(close, fastLength, slowLength, signalLength)

// Plot
plot(macdLine, color=color.blue, title="MACD")
plot(signalLine, color=color.orange, title="Signal")
plot(histLine, style=plot.style_histogram, color=histLine > 0 ? color.green : color.red, title="Histogram")

// Generate signals
longCondition = ta.crossover(macdLine, signalLine) and macdLine < 0
shortCondition = ta.crossunder(macdLine, signalLine) and macdLine > 0

if (longCondition)
    strategy.entry("Long", strategy.long)
    alert('{"secret":"YOUR_SECRET","action":"buy","coin":"' + syminfo.ticker + '","leverage":10,"collateral_usd":100,"indicator":"macd"}', alert.freq_once_per_bar)

if (shortCondition)
    strategy.entry("Short", strategy.short)
    alert('{"secret":"YOUR_SECRET","action":"sell","coin":"' + syminfo.ticker + '","leverage":10,"collateral_usd":100,"indicator":"macd"}', alert.freq_once_per_bar)
